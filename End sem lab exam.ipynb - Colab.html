import pandas as pd import matplotlib.pyplot as plt from sklearn.linear_model import LinearRegression from sklearn.metrics import r2_score df=pd.DataFrame({ 'Rainfall':[400,600,300,700,500], 'Yield':[2.5,3.8,1.8,4.5,3.1] }) X=df[['Rainfall']] y=df['Yield'] print("Correlation:",df.corr().loc['Rainfall','Yield']) m=LinearRegression().fit(X,y) p=m.predict(X) print("550 mm Yield:",m.predict([[550]])[0]) print("R2:",r2_score(y,p)) plt.scatter(X,y) plt.plot(X,p) plt.show() plt.scatter(p,y-p) plt.axhline(0) plt.show()
